## Integration with Gameplay

### Apply Stats to Player

```csharp
// In PlayerController.Start() or VehicleManager
void ApplyVehicleStats()
{
    if (VehicleUnlockSystem.Instance == null) return;

    VehicleStats stats = VehicleUnlockSystem.Instance.GetCurrentStats();

    // Apply speed
    moveSpeed *= stats.speedMultiplier;

    // Apply handling
    laneChangeSpeed *= stats.handlingMultiplier;

    // Store other stats for runtime use
    this.coinMultiplier = stats.coinMultiplier;
    this.magnetRange *= stats.magnetRangeMultiplier;
    this.shieldDuration *= stats.shieldDurationMultiplier;

    // Grant starting shield
    if (stats.startingShieldDuration > 0)
    {
        ActivateShield(stats.startingShieldDuration);
    }

    Debug.Log($"Vehicle stats applied: {VehicleUnlockSystem.Instance.GetSelectedVehicle().vehicleName}");
}
```

### Handle Lucky Dodge

```csharp
// In Obstacle.OnTriggerEnter()
void OnTriggerEnter(Collider other)
{
    if (other.CompareTag("Player"))
    {
        // Check lucky dodge
        var stats = VehicleUnlockSystem.Instance?.GetCurrentStats();
        if (stats != null && stats.luckyDodgeChance > 0)
        {
            float roll = Random.Range(0f, 100f);
            if (roll < stats.luckyDodgeChance)
            {
                Debug.Log("Lucky Dodge! Obstacle avoided!");
                ShowLuckyDodgeEffect();
                return; // Don't trigger game over
            }
        }

        // Normal collision
        GameOver();
    }
}
```

### Handle Extra Life

```csharp
// In GameStateManager or PlayerController
private bool hasUsedExtraLife = false;

void OnCollision()
{
    var stats = VehicleUnlockSystem.Instance?.GetCurrentStats();

    if (stats != null && stats.extraLife > 0 && !hasUsedExtraLife)
    {
        // Use extra life
        hasUsedExtraLife = true;
        Debug.Log("Extra Life Used! Continue playing...");

        // Grant temporary shield
        ActivateShield(3f);

        // Show UI notification
        ShowExtraLifeNotification();

        return; // Don't game over
    }

    // Normal game over
    GameOver();
}
```
